#legend
| : OR
? : 0..1
* : 0..n
##

program: (statement ';')* EOI

statement: vaDecl
         | expr
		 
varDecl: VAR IDENTIFIER '=' expr

expr: mathExpr ('?' exp ':' exp)?

mathExpr: term (('+' | '-') term)*

term: factor (('*' | '/' | '%') factor)*

factor: '-'? positiveFactor
	  
positiveFactor: NUMBER
              | IDENTIFIER
              | '(' expr ')'
